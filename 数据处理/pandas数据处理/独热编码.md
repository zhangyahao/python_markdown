 ```
 data = {
    '年龄': np.random.randint(18, 65, size=19),
    '性别': np.random.choice(['男', '女'], size=19),
    '尺码': np.random.choice(['S', 'M', 'L', 'XL'], size=19, p=[0.2, 0.5, 0.2, 0.1])
}

df = pd.DataFrame(data)
df
gender=pd.get_dummies(df['性别'],prefix='性别',dtype=int)
df=pd.concat(objs=[df,gender],axis=1)
 ```
